async function importarAbarrotes(archivoExcel) {

        name: row['Descripcion'],
        barcode: row['Codigo_Barras'],
        description: row['Marca'],
        categoryId: 'CATEGORIA_ID',
        companyId: 'COMPANY_ID',
        price: 0,
        cost: 0,
        stock: 0
      }
    })
  }
